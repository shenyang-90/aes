
### Command Used
```bash

    --coverage-line --coverage-toggle \
    --public-flat-rw \
    --top-module tb_coverage \
    [RTL files] tb_coverage.sv sim_main_coverage.cpp

# Run simulation
./Vtb_coverage

# Generate report
verilator_coverage --write-info coverage.info coverage.dat
genhtml coverage.info -o html_report/
```
