### Generation Steps
```bash
# 1. Compile with Verilator
verilator --cc --trace --timing --coverage-line --coverage-toggle \
  --public-flat-rw -Wno-PINMISSING -Wno-WIDTHTRUNC -Wno-WIDTHEXPAND \
  -Mdir obj_dir -CFLAGS "-std=c++20 -O2" -LDFLAGS "-lpthread" \
  --build --exe --top-module tb_coverage \
  Database/RTL/*.v \
  Database/Verification/Env/verilator/tb_coverage.sv \
  Database/Verification/Env/verilator/sim_main.cpp

# 2. Run simulation
./obj_dir/Vtb_coverage +trace

# 3. Convert coverage data
verilator_coverage --write-info coverage.info coverage.dat

# 4. Generate HTML
genhtml coverage.info -o html/ --ignore-errors source
```
